748  void setRotation(const Vector& axis, float angleRadians) {
749  const Vector n = axis.normalized();
750  const float s = std::sin(angleRadians);
751  const float c = std::cos(angleRadians);
752  const float C = (1-c);
753 
754  xBasis = Vector(n[0]*n[0]*C + c, n[0]*n[1]*C - n[2]*s, n[0]*n[2]*C + n[1]*s);
755  yBasis = Vector(n[1]*n[0]*C + n[2]*s, n[1]*n[1]*C + c, n[1]*n[2]*C - n[0]*s);
756  zBasis = Vector(n[2]*n[0]*C - n[1]*s, n[2]*n[1]*C + n[0]*s, n[2]*n[2]*C + c );
757  }
